<RequestExample>

```python python
from retab import Retab

client = Retab()

# Split a multi-page document into sections based on subdocuments
# Use partition_key to identify individual items within a subdocument
# Use n_consensus when you want consensus-based confidence signals
result = client.documents.split(
    document="invoice_batch.pdf",
    model="retab-small",
    subdocuments=[
        {"name": "invoice", "description": "Invoice documents with billing information", "partition_key": "invoice number"},
        {"name": "receipt", "description": "Receipt documents for payments"},
        {"name": "contract", "description": "Legal contract documents"},
    ],
    context="Processing Q4 2024 vendor document batch",  # Optional: additional context
    n_consensus=3
)

# Access split results
for split in result.splits:
    print(f"{split.name}: pages {split.pages}")
    # Access partitions when partition_key is specified
    for partition in split.partitions:
        print(f"  {partition.key}: pages {partition.pages}")
print(result.consensus.likelihoods if result.consensus else None)
# choices[0] is the consolidated split output, choices[1:] are the raw consensus runs
print(len(result.consensus.choices) if result.consensus else 0)

# Example output:
# invoice: pages [1, 2, 3]
#   INV-001: pages [1, 2]
#   INV-002: pages [3]
# receipt: pages [4, 5]
# contract: pages [6, 7, 8]
```

```typescript typescript
import { Retab, type SplitRequest, type SplitResponse } from '@retab/node';
import { config } from 'dotenv';

config();

const client = new Retab();

const splitRequest: SplitRequest = {
    document: "invoice_batch.pdf",
    model: "retab-small",
    subdocuments: [
        { name: "invoice", description: "Invoice documents with billing information", partition_key: "invoice number" },
        { name: "receipt", description: "Receipt documents for payments" },
        { name: "contract", description: "Legal contract documents" },
    ],
    context: "Processing Q4 2024 vendor document batch",  // Optional: additional context
    n_consensus: 3
};

const result: SplitResponse = await client.documents.split(splitRequest);

// Access split results
result.splits.forEach(split => {
    console.log(`${split.name}: pages ${split.pages}`);
    // Access partitions when partition_key is specified
    split.partitions.forEach(partition => {
        console.log(`  ${partition.key}: pages ${partition.pages}`);
    });
});
console.log(result.consensus?.likelihoods);
// choices[0] is the consolidated split output, choices[1:] are the raw consensus runs
console.log(result.consensus?.choices.length ?? 0);
```
