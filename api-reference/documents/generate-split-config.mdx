---
title: 'Generate Split Config'
openapi: 'POST /v1/documents/split/generate_config'
---

<RequestExample>

```python python
from retab import Retab

client = Retab()

result = client.documents.generate_split_config(
    document="property_portfolio.pdf",
    model="retab-small",
)

for subdocument in result.subdocuments:
    print(subdocument.name, subdocument.description, subdocument.partition_key)
```

```javascript javascript
import { Retab } from '@retab/node';
import { config } from 'dotenv';

config();

const client = new Retab();

const result = await client.documents.generate_split_config({
    document: "property_portfolio.pdf",
    model: "retab-small",
});

for (const subdocument of result.subdocuments) {
    console.log(subdocument.name, subdocument.description, subdocument.partition_key);
}
```

```typescript typescript
import { Retab, type GenerateSplitConfigRequest, type GenerateSplitConfigResponse } from '@retab/node';
import { config } from 'dotenv';

config();

const client = new Retab();

const request: GenerateSplitConfigRequest = {
    document: "property_portfolio.pdf",
    model: "retab-small",
};

const result: GenerateSplitConfigResponse = await client.documents.generate_split_config(request);

for (const subdocument of result.subdocuments) {
    console.log(subdocument.name, subdocument.description, subdocument.partition_key);
}
```
