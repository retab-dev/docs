```python python
reclient = Retab()
```

```bash curl
curl -X 'POST' \
  'https://api.retab.com/v1/documents/correct_image_orientation' \
  -H "Api-Key: $RETAB_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
  "document": {
    "id": "string",
    "name": "string",
    "mime_type": "string",
    "content": "string"
  },
}'
```
