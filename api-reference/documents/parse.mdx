<RequestExample>

```python python
from retab import Retab

client = Retab()
result = client.documents.parse(
    document="document.pdf",
    model="retab-small",
    table_parsing_format="markdown",
    image_resolution_dpi=192,
)

# Access parsed content
for i, page_content in enumerate(result.pages):
    print(f"Page {i + 1}: {page_content}")

print(f"Filename: {result.document.filename}")
print(f"Total pages: {result.usage.page_count}")
print(f"Credits used: {result.usage.credits}")
```

```javascript javascript
import { Retab } from '@retab/node';
import { config } from 'dotenv';

config();

const client = new Retab();

const result = await client.documents.parse({
    document: "document.pdf",
    model: "retab-small",
    table_parsing_format: "markdown",
    image_resolution_dpi: 192,
});

// Access parsed content
result.pages.forEach((pageContent, index) => {
    console.log(`Page ${index + 1}: ${pageContent}`);
});

console.log(`Filename: ${result.document.filename}`);
console.log(`Total pages: ${result.usage.page_count}`);
console.log(`Credits used: ${result.usage.credits}`);
```

```typescript typescript
import { Retab, type ParseRequest, type ParseResponse } from '@retab/node';
import { config } from 'dotenv';

config();

const client = new Retab();

const parseRequest: ParseRequest = {
    document: "document.pdf",
    model: "retab-small",
    table_parsing_format: "markdown",
    image_resolution_dpi: 192,
};

const result: ParseResponse = await client.documents.parse(parseRequest);

// Access parsed content
result.pages.forEach((pageContent: string, index: number) => {
    console.log(`Page ${index + 1}: ${pageContent}`);
});

console.log(`Filename: ${result.document.filename}`);
console.log(`Total pages: ${result.usage.page_count}`);
console.log(`Credits used: ${result.usage.credits}`);
```
