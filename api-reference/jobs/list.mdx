```go Go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func ptr[T any](v T) *T { return &v }

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	// List all jobs
	jobsResponse, err := client.Jobs.List(ctx, nil)
	if err != nil {
		log.Fatal(err)
	}
	for _, job := range jobsResponse.Data {
		fmt.Printf("%s: %s - %s\n", job.ID, *job.Status, job.Endpoint)
	}

	// List with filters and pagination
	status := retab.JobStatusCompleted
	filtered, err := client.Jobs.List(ctx, &retab.JobsListParams{
		PaginationParams: retab.PaginationParams{Limit: ptr(10)},
		Status:           &status,
	})
	if err != nil {
		log.Fatal(err)
	}

	// Paginate through results
	if filtered.ListMetadata.After != "" {
		nextPage, err := client.Jobs.List(ctx, &retab.JobsListParams{
			PaginationParams: retab.PaginationParams{
				After: ptr(filtered.ListMetadata.After),
				Limit: ptr(10),
			},
		})
		if err != nil {
			log.Fatal(err)
		}
		_ = nextPage
	}
}
```

```rust Rust
use retab::enums::JobsStatus;
use retab::resources::jobs::ListParams;
use retab::Retab;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let client = Retab::new(std::env::var("RETAB_API_KEY")?);

    // List all jobs
    let jobs_response = client.jobs().list(ListParams::default()).await?;
    for job in &jobs_response.data {
        println!(
            "{}: {:?} - {}",
            job.id,
            job.status,
            job.endpoint
        );
    }

    // List with filters and pagination
    let filtered = client
        .jobs()
        .list(ListParams {
            status: Some(JobsStatus::Completed),
            limit: Some(10),
            ..Default::default()
        })
        .await?;

    // Paginate through results
    if let Some(after) = filtered.list_metadata.after.clone() {
        let _next_page = client
            .jobs()
            .list(ListParams {
                after: Some(after),
                limit: Some(10),
                ..Default::default()
            })
            .await?;
    }
    Ok(())
}
```

## Pagination

Jobs are returned in descending order by creation time (newest first). Use id-based pagination to iterate through all results:

<CodeGroup>
```python Python
all_jobs = []
after = None

while True:
response = client.jobs.list(after=after, limit=100)
all_jobs.extend(response.data)

    if not response.has_more:
        break
    after = response.list_metadata.after

print(f"Total jobs: {len(all_jobs)}")

````

```typescript TypeScript
const allJobs = [];
let after = undefined;

while (true) {
    const response = await client.jobs.list({ after, limit: 100 });
    allJobs.push(...response.data);

    if (!response.list_metadata.after) {
        break;
    }
    after = response.list_metadata.after;
}

console.log(`Total jobs: ${allJobs.length}`);
````

```go Go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func ptr[T any](v T) *T { return &v }

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	allJobs := []retab.Job{}
	after := ""

	for {
		response, err := client.Jobs.List(ctx, &retab.JobsListParams{
			PaginationParams: retab.PaginationParams{
				After: ptr(after),
				Limit: ptr(100),
			},
		})
		if err != nil {
			log.Fatal(err)
		}
		allJobs = append(allJobs, response.Data...)

		if response.ListMetadata.After == "" {
			break
		}
		after = response.ListMetadata.After
	}

	fmt.Printf("Total jobs: %d\n", len(allJobs))
}
```

### Filter by Metadata

While the API doesn't support direct metadata filtering, you can filter client-side:

<CodeGroup>
```python Python
# Find jobs from a specific batch
batch_id = "batch_001"
batch_jobs = []

response = client.jobs.list(limit=100)
for job in response.data:
if job.metadata and job.metadata.get("batch_id") == batch_id:
batch_jobs.append(job)

print(f"Jobs in {batch_id}: {len(batch_jobs)}")

````

```typescript TypeScript
// Find jobs from a specific batch
const batchId = "batch_001";
const batchJobs = [];

const response = await client.jobs.list({ limit: 100 });
for (const job of response.data) {
    if (job.metadata?.batch_id === batchId) {
        batchJobs.push(job);
    }
}

console.log(`Jobs in ${batchId}: ${batchJobs.length}`);
````

```go Go
ctx := context.Background()
client, _ := retab.NewClient("")

// Find jobs from a specific batch
batchID := "batch_001"
batchJobs := []retab.Job{}

response, err := client.Jobs.List(ctx, &retab.JobsListParams{Limit: ptr(100)})
if err != nil {
	log.Fatal(err)
}
for _, job := range response.Data {
	metadata, ok := job["metadata"].(map[string]any)
	if !ok {
		continue
	}
	if metadata["batch_id"] == batchID {
		batchJobs = append(batchJobs, job)
	}
}

fmt.Printf("Jobs in %s: %d\n", batchID, len(batchJobs))
```

```ruby Ruby
# Find jobs from a specific batch

  if job.metadata && job.metadata['batch_id'] == batch_id

```

```rust Rust
use retab::resources::jobs::ListParams;
```
