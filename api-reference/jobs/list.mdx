List legacy jobs for the authenticated organization.

<Note>
  The generic Jobs API is documented for existing integrations. New SDK clients should use background mode on primitive create endpoints instead.
</Note>

```python Python
import os
import urllib.request

request = urllib.request.Request(
    "https://api.retab.com/v1/jobs",
    method="GET",
    headers={"Authorization": f"Bearer {os.environ['RETAB_API_KEY']}"},
)
with urllib.request.urlopen(request) as response:
    print(response.status)
```

```go Go
package main

import (
    "fmt"
    "net/http"
    "os"
)

func main() {
    request, err := http.NewRequest("GET", "https://api.retab.com/v1/jobs", nil)
    if err != nil {
        panic(err)
    }
    request.Header.Set("Authorization", "Bearer "+os.Getenv("RETAB_API_KEY"))
    response, err := http.DefaultClient.Do(request)
    if err != nil {
        panic(err)
    }
    defer response.Body.Close()
    fmt.Println(response.StatusCode)
}
```

```java Java
import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;

public final class Example {
  public static void main(String[] args) throws Exception {
    HttpRequest request = HttpRequest.newBuilder()
        .uri(URI.create("https://api.retab.com/v1/jobs"))
        .header("Authorization", "Bearer " + System.getenv("RETAB_API_KEY"))
        .method("GET", java.net.http.HttpRequest.BodyPublishers.noBody())
        .build();

    HttpResponse<String> response = HttpClient.newHttpClient()
        .send(request, HttpResponse.BodyHandlers.ofString());
    System.out.println(response.statusCode());
  }
}
```
