```go Go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func ptr[T any](v T) *T { return &v }

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	// Create an async extraction job
	job, err := client.Jobs.Create(ctx, &retab.JobsCreateParams{
		Endpoint: retab.CreateJobRequestEndpointV1Extractions,
		Request: retab.Resource{
			"document": map[string]any{
				"filename": "invoice.pdf",
				"url":      "data:application/pdf;base64,JVBERi0xLjQK...",
			},
			"json_schema": map[string]any{
				"type": "object",
				"properties": map[string]any{
					"invoice_number": map[string]any{"type": "string"},
					"total":          map[string]any{"type": "number"},
				},
				"required": []string{"invoice_number", "total"},
			},
			"model": "retab-small",
		},
		Metadata: map[string]string{"batch_id": "batch_001", "source": "api"},
	})
	if err != nil {
		log.Fatal(err)
	}

	fmt.Printf("Job ID: %s\n", job.ID)
	fmt.Printf("Status: %s\n", *job.Status)
}
```

```rust Rust
use retab::enums::CreateJobRequestEndpoint;
use retab::models::CreateJobRequest;
use retab::resources::jobs::CreateParams;
use retab::Retab;
use serde_json::json;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let client = Retab::new(std::env::var("RETAB_API_KEY")?);

    let request = json!({
        "document": {
            "filename": "invoice.pdf",
            "url": "data:application/pdf;base64,JVBERi0xLjQK...",
        },
        "json_schema": {
            "type": "object",
            "properties": {
                "invoice_number": {"type": "string"},
                "total": {"type": "number"},
            },
            "required": ["invoice_number", "total"],
        },
        "model": "retab-small",
    });
    let request_map = request.as_object().unwrap().clone().into_iter().collect();

    let body = CreateJobRequest {
        endpoint: CreateJobRequestEndpoint::V1Extractions,
        request: request_map,
        metadata: Some(
            [
                ("batch_id".to_string(), "batch_001".to_string()),
                ("source".to_string(), "api".to_string()),
            ]
            .into_iter()
            .collect(),
        ),
    };

    let job = client.jobs().create(CreateParams::new(body)).await?;
    println!("Job ID: {}", job.id);
    println!("Status: {:?}", job.status);
    Ok(())
}
```

## Dispatch Behavior

Most successful creates return `200` with a `queued` job. If the job record is stored successfully but task dispatch is temporarily delayed, Retab returns `202 Accepted` with the job already created and a warning like `dispatch_delayed`. In that case, poll the returned `job.id` instead of creating a duplicate job.
