```rust Rust
                include_response: Some(true),

        )
        .await?;

        Some(JobStatus::Completed) => {

        Some(JobStatus::Failed) => {
            if let Some(err) = &job.error {
                println!("Error: {:?}", err);
            }
        }
        _ => {}
    }
    Ok(())
}
```

## SDK Convenience Methods

- Python: `client.jobs.get(job_id)` defaults to lightweight polling. Pass `include_request=True` or `include_response=True` when you need payloads.
- JavaScript: `client.jobs.get(jobId)` defaults to lightweight polling. Pass `include_request: true` or `include_response: true` when you need payloads.
- Go: `client.Jobs.Retrieve(ctx, jobID, params)` defaults to lightweight polling when `params` is nil. Pass `JobRetrieveParams` when you need payloads.

## Payload Restoration

`include_request=true` restores any request-side file artifacts that were offloaded during job creation.

`include_response=true` restores response payloads or document artifacts for completed jobs.

If restoration fails for a specific artifact, the job is still returned and Retab may attach a warning describing the degraded restore.

## Polling for Completion

Use polling to wait for a job to complete:

<CodeGroup>
```python Python
import time

job = client.jobs.get(job_id)
terminal_statuses = ("completed", "failed", "cancelled", "expired")

while job.status not in terminal_statuses:
    time.sleep(2) # Wait 2 seconds between polls
    job = client.jobs.get(job_id)

# Fetch the payload only once the job is terminal

if job.status == "completed":
    job = client.jobs.get(job_id, include_response=True)

# Or use the helper method

job = client.jobs.wait_for_completion(
    job_id,
    poll_interval_seconds=2.0,
    timeout_seconds=300,
)

````

```typescript TypeScript
const terminalStatuses = ["completed", "failed", "cancelled", "expired"];

while (!terminalStatuses.includes(job.status)) {
    await new Promise(r => setTimeout(r, 2000));  // Wait 2 seconds
    job = await client.jobs.get(jobId);
}

if (job.status === "completed") {
    job = await client.jobs.get(jobId, { includeResponse: true });
}

// Or use the helper method
job = await client.jobs.waitForCompletion(jobId, {

    timeout_ms: 300_000,
});
````

```go Go
package main

import (
	"context"
	"fmt"
	"log"
	"time"

	retab "github.com/retab-dev/retab/clients/go"
)

func ptr[T any](v T) *T { return &v }

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	jobID := "job_V1StGXR8_Z5jdHi6B-myT"

	// Manual polling
	terminal := map[string]bool{"completed": true, "failed": true, "cancelled": true, "expired": true}
	job, err := client.Jobs.Get(ctx, jobID, nil)
	if err != nil {
		log.Fatal(err)
	}
	for {
		status := string(*job.Status)
		if terminal[status] {
			break
		}
		time.Sleep(2 * time.Second)
		job, err = client.Jobs.Get(ctx, jobID, nil)
		if err != nil {
			log.Fatal(err)
		}
	}

	if status := string(*job.Status); status == "completed" {
		job, err = client.Jobs.Get(ctx, jobID, &retab.JobsGetParams{IncludeResponse: ptr(true)})
		if err != nil {
			log.Fatal(err)
		}
	}

	fmt.Println(*job.Status)
}
```

```rust Rust
use retab::enums::JobStatus;
use retab::resources::jobs::GetParams;
use retab::Retab;
use std::time::Duration;

let client = Retab::new(std::env::var("RETAB_API_KEY")?);

let job_id = "job_V1StGXR8_Z5jdHi6B-myT";
let mut job = client.jobs().get(job_id, GetParams::default()).await?;

while !matches!(
    job.status,
    Some(JobStatus::Completed | JobStatus::Failed | JobStatus::Cancelled | JobStatus::Expired)
) {
    tokio::time::sleep(Duration::from_secs(2)).await;
    job = client.jobs().get(job_id, GetParams::default()).await?;
}

// Fetch the payload only once the job is terminal.
if job.status == Some(JobStatus::Completed) {
    job = client.jobs().get(job_id, GetParams {
        include_response: Some(true),
        ..Default::default()
    }).await?;
}
```
