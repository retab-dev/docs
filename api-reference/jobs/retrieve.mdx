```python python
from retab import Retab

client = Retab()

# Retrieve a job by ID
job = client.jobs.retrieve(

    include_request=True,
    include_response=True,
)
```

```go go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	// Retrieve a job by ID
	job, err := client.Jobs.Retrieve(ctx, "job_V1StGXR8_Z5jdHi6B-myT", &retab.JobRetrieveParams{
		IncludeRequest:  true,
		IncludeResponse: true,
	})
	if err != nil {
		log.Fatal(err)
	}

	status, _ := (*job)["status"].(string)
	fmt.Printf("Status: %s\n", status)

	switch status {
	case "completed":
		if response, ok := (*job)["response"].(map[string]any); ok {
			fmt.Println("Result:", response["body"])
		}
	case "failed":
		if jobErr, ok := (*job)["error"].(map[string]any); ok {
			fmt.Println("Error:", jobErr["message"])
		}
	}
}
```

## SDK Convenience Methods

- Python: `client.jobs.retrieve(job_id)` defaults to lightweight polling. Pass `include_request=True` or `include_response=True` when you need payloads.
- JavaScript: `client.jobs.retrieve(jobId)` defaults to lightweight polling. Pass `include_request: true` or `include_response: true` when you need payloads.
- Go: `client.Jobs.Retrieve(ctx, jobID, params)` defaults to lightweight polling when `params` is nil. Pass `JobRetrieveParams` when you need payloads.

## Payload Restoration

`include_request=true` restores any request-side file artifacts that were offloaded during job creation.

`include_response=true` restores response payloads or document artifacts for completed jobs.

If restoration fails for a specific artifact, the job is still returned and Retab may attach a warning describing the degraded restore.

## Polling for Completion

Use polling to wait for a job to complete:

<CodeGroup>
```python Python
import time

job = client.jobs.retrieve(job_id)
terminal_statuses = ("completed", "failed", "cancelled", "expired")

while job.status not in terminal_statuses:
    time.sleep(2) # Wait 2 seconds between polls
    job = client.jobs.retrieve(job_id)

# Fetch the payload only once the job is terminal

if job.status == "completed":
    job = client.jobs.retrieve(job_id, include_response=True)

# Or use the helper method

job = client.jobs.wait_for_completion(
    job_id,
    poll_interval_seconds=2.0,
    timeout_seconds=300,
)

````

```javascript JavaScript
const terminalStatuses = ["completed", "failed", "cancelled", "expired"];

while (!terminalStatuses.includes(job.status)) {
    await new Promise(r => setTimeout(r, 2000));  // Wait 2 seconds
    job = await client.jobs.retrieve(jobId);
}

// Fetch the payload only once the job is terminal
if (job.status === "completed") {
    job = await client.jobs.retrieve(jobId, { include_response: true });
}

// Or use the helper method
job = await client.jobs.waitForCompletion(jobId, {
    poll_interval_ms: 2000,
    timeout_ms: 300_000,
});
````

```go Go
package main

import (
	"context"
	"fmt"
	"log"
	"time"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	jobID := "job_V1StGXR8_Z5jdHi6B-myT"

	// Manual polling
	terminal := map[string]bool{"completed": true, "failed": true, "cancelled": true, "expired": true}
	job, err := client.Jobs.Retrieve(ctx, jobID, nil)
	if err != nil {
		log.Fatal(err)
	}
	for {
		status, _ := (*job)["status"].(string)
		if terminal[status] {
			break
		}
		time.Sleep(2 * time.Second)
		job, err = client.Jobs.Retrieve(ctx, jobID, nil)
		if err != nil {
			log.Fatal(err)
		}
	}

	if status, _ := (*job)["status"].(string); status == "completed" {
		job, err = client.Jobs.Retrieve(ctx, jobID, &retab.JobRetrieveParams{IncludeResponse: true})
		if err != nil {
			log.Fatal(err)
		}
	}

	// Or use the helper method
	job, err = client.Jobs.WaitForCompletion(ctx, jobID, &retab.JobWaitForCompletionParams{
		PollInterval:    2 * time.Second,
		Timeout:         5 * time.Minute,
		IncludeResponse: true,
	})
	if err != nil {
		log.Fatal(err)
	}
	fmt.Println((*job)["status"])
}
```

</CodeGroup>

By default, `wait_for_completion`/`waitForCompletion` polls with lightweight retrieves and returns a final job with `include_response=true` and `include_request=false`. Override these with options when needed.
