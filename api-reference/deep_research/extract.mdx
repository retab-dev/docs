---
title: 'Extract'
openapi: 'POST /v1/beta/deepresearch/extract'
---

Run a deep-research extraction — the LLM reads the document, generates
intermediate research steps, and emits a final structured JSON payload that
matches the supplied schema.

<Warning>
Beta endpoint. The request and response shapes may change without notice.
There is no dedicated SDK helper yet — use the HTTP client examples below.
</Warning>

<RequestExample>

```python python
import httpx
import os

response = httpx.post(
    "https://api.retab.com/v1/beta/deepresearch/extract",
    headers={
        "Api-Key": os.environ["RETAB_API_KEY"],
        "Content-Type": "application/json",
    },
    json={
        "model": "claude-opus-4-7",
        "json_schema": {
            "type": "object",
            "properties": {
                "summary": {"type": "string"},
                "key_findings": {"type": "array", "items": {"type": "string"}},
            },
        },
        "document": {
            "filename": "report.pdf",
            "url": "data:application/pdf;base64,<BASE64>",
        },
    },
    timeout=600.0,
)
response.raise_for_status()
print(response.json())
```

```go go
package main

import (
	"bytes"
	"encoding/json"
	"fmt"
	"io"
	"log"
	"net/http"
	"os"
)

func main() {
	body, err := json.Marshal(map[string]any{
		"model": "claude-opus-4-7",
		"json_schema": map[string]any{
			"type": "object",
			"properties": map[string]any{
				"summary":      map[string]any{"type": "string"},
				"key_findings": map[string]any{"type": "array", "items": map[string]any{"type": "string"}},
			},
		},
		"document": map[string]any{
			"filename": "report.pdf",
			"url":      "data:application/pdf;base64,<BASE64>",
		},
	})
	if err != nil {
		log.Fatal(err)
	}

	request, err := http.NewRequest(
		http.MethodPost,
		"https://api.retab.com/v1/beta/deepresearch/extract",
		bytes.NewReader(body),
	)
	if err != nil {
		log.Fatal(err)
	}
	request.Header.Set("Api-Key", os.Getenv("RETAB_API_KEY"))
	request.Header.Set("Content-Type", "application/json")

	response, err := http.DefaultClient.Do(request)
	if err != nil {
		log.Fatal(err)
	}
	defer response.Body.Close()

	responseBody, err := io.ReadAll(response.Body)
	if err != nil {
		log.Fatal(err)
	}
	if response.StatusCode >= 400 {
		log.Fatalf("request failed: %s", responseBody)
	}

	var result map[string]any
	if err := json.Unmarshal(responseBody, &result); err != nil {
		log.Fatal(err)
	}
	fmt.Println(result)
}
```
