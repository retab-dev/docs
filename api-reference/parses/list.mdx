---
title: "List Parses"
openapi: "GET /v1/parses"
---

List persisted parses for your organization, with id-based pagination and optional filename / date filtering.

<RequestExample>

```python Python
from datetime import datetime
from retab import Retab

client = Retab()

# List recent parses
parses = client.parses.list(limit=20, order="desc")
for parse in parses.data:
    print(f"{parse.id}: {parse.file.filename}")

# Filter by date range
parses = client.parses.list(
    from_date=datetime(2024, 1, 1),
    to_date=datetime(2024, 12, 31),
    limit=50,
)
```

```go Go
package main

import (
	"context"
	"fmt"
	"log"
	"time"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	// List recent parses
	parses, err := client.Parses.List(ctx, &retab.ListParams{Limit: 20, Order: "desc"})
	if err != nil {
		log.Fatal(err)
	}
	for _, parse := range parses.Data {
		fmt.Printf("%v: %v\n", parse["id"], parse["file"])
	}

	// Filter by date range
	from := time.Date(2024, 1, 1, 0, 0, 0, 0, time.UTC)
	to := time.Date(2024, 12, 31, 0, 0, 0, 0, time.UTC)
	ranged, err := client.Parses.List(ctx, &retab.ListParams{
		FromDate: &from,
		ToDate:   &to,
		Limit:    50,
	})
	if err != nil {
		log.Fatal(err)
	}
	fmt.Println(ranged)
}
```

```rust Rust
use retab::enums::ParsesOrder;
use retab::resources::parses::ListParams;
use retab::Retab;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let client = Retab::new(std::env::var("RETAB_API_KEY")?);

    // List recent parses
    let parses = client
        .parses()
        .list(ListParams {
            limit: Some(20),
            order: Some(ParsesOrder::Desc),
            ..Default::default()
        })
        .await?;
    for parse in &parses.data {
        println!("{}: {}", parse.id, parse.file.filename);
    }

    // Filter by date range
    let _ranged = client
        .parses()
        .list(ListParams {
            from_date: Some("2024-01-01".into()),
            to_date: Some("2024-12-31".into()),
            limit: Some(50),
            ..Default::default()
        })
        .await?;
    Ok(())
}
```
