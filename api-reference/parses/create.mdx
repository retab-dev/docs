---
title: "Create Parse"
openapi: "POST /v1/parses"
---

Parse a document into normalized text and persist the result as a `Parse` resource that can later be retrieved via `GET /v1/parses/{parse_id}` or listed via `GET /v1/parses`.

<RequestExample>

```python Python
from retab import MIMEData, Retab

client = Retab()

document = MIMEData(
    filename="document.pdf",
    url="https://my-bucket.s3.us-east-1.amazonaws.com/documents/document.pdf",
)

parse = client.parses.create(
    document=document,
    model="retab-small",
    table_parsing_format="markdown",
)

print(f"Parse ID: {parse.id}")
print(f"Filename: {parse.file.filename}")
print(f"Full text: {parse.output.text}")
for i, page in enumerate(parse.output.pages):
    print(f"Page {i + 1}: {page}")
```

```go Go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func ptr[T any](v T) *T { return &v }

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	document := retab.MIMEData{
		Filename: "document.pdf",
		URL:      "https://my-bucket.s3.us-east-1.amazonaws.com/documents/document.pdf",
	}

	parse, err := client.Parses.Create(ctx, &retab.ParsesCreateParams{
		Document:           document,
		Model:              ptr("retab-small"),
		TableParsingFormat: ptr(retab.ParseRequestTableParsingFormatMarkdown),
	})
	if err != nil {
		log.Fatal(err)
	}

	fmt.Printf("Parse ID: %s\n", parse.ID)
	fmt.Printf("Full text: %s\n", parse.Output.Text)
	for i, page := range parse.Output.Pages {
		fmt.Printf("Page %d: %s\n", i+1, page)
	}
}
```

```rust Rust
use retab::enums::ParseRequestTableParsingFormat;
use retab::resources::parses::CreateParams;
use retab::{MimeData, Retab};

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let client = Retab::new(std::env::var("RETAB_API_KEY")?);
    let document = MimeData::new(
        "document.pdf",
        "https://my-bucket.s3.us-east-1.amazonaws.com/documents/document.pdf",
    );

    let mut params = CreateParams::new(document);
    params.body.model = Some("retab-small".into());
    params.body.table_parsing_format = Some(ParseRequestTableParsingFormat::Markdown);

    let parse = client.parses().create(params).await?;

    println!("Parse ID: {}", parse.id);
    println!("Filename: {}", parse.file.filename);
    println!("Full text: {}", parse.output.text);
    for (i, page) in parse.output.pages.iter().enumerate() {
        println!("Page {}: {}", i + 1, page);
    }
    Ok(())
}
```

```java Java
import com.retab.RetabClient;
import com.retab.models.MimeData;
import com.retab.types.ParseRequestTableParsingFormat;
import java.net.URI;

public final class Example {
  public static void main(String[] args) throws Exception {
    RetabClient client = new RetabClient(System.getenv("RETAB_API_KEY"));

    MimeData document = MimeData.fromUrl(
        URI.create("https://my-bucket.s3.us-east-1.amazonaws.com/documents/document.pdf"));

    var result =
        client
            .parses()
            .create(
                document,
                "retab-small",
                ParseRequestTableParsingFormat.MARKDOWN,
                null,
                null,
                null);
    System.out.println(result);
  }
}
```
