---
title: 'Update Extraction'
openapi: 'PATCH /v1/extractions/{extraction_id}'
---

<RequestExample>

```python python
from retab import Retab

client = Retab()

# Update predictions after human review
updated = client.extractions.update(
    extraction_id="extr_01G34H8J2K",
    predictions={
        "invoice_number": "INV-2024-0789-CORRECTED",
        "total_amount": 1576.75
    },
    human_review_status="reviewed"
)

# Update only the review status
updated = client.extractions.update(
    extraction_id="extr_01G34H8J2K",
    human_review_status="reviewed"
)

# Update metadata
updated = client.extractions.update(
    extraction_id="extr_01G34H8J2K",
    metadata={"reviewed_by": "john@example.com"}
)
```

```typescript typescript
import { Retab } from "@retab/node";

const client = new Retab();

// Update predictions after human review
const updated = await client.extractions.update("extr_01G34H8J2K", {
    predictions: {
        invoice_number: "INV-2024-0789-CORRECTED",
        total_amount: 1576.75
    },
    human_review_status: "reviewed"
});

// Update only the review status
const updated = await client.extractions.update("extr_01G34H8J2K", {
    human_review_status: "reviewed"
});

// Update metadata
const updated = await client.extractions.update("extr_01G34H8J2K", {
    metadata: { reviewed_by: "john@example.com" }
});
```
