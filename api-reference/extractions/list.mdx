```go Go
package main

import (
	"context"
	"fmt"
	"log"
	"time"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	// List recent extractions
	extractions, err := client.Extractions.List(ctx, &retab.ListExtractionsParams{
		ListParams: retab.ListParams{Limit: 10, Order: "desc"},
	})
	if err != nil {
		log.Fatal(err)
	}
	fmt.Println(extractions)

	// Filter by date range
	from := time.Date(2024, 1, 1, 0, 0, 0, 0, time.UTC)
	to := time.Date(2024, 12, 31, 0, 0, 0, 0, time.UTC)
	extractions, err = client.Extractions.List(ctx, &retab.ListExtractionsParams{
		ListParams: retab.ListParams{
			Limit:    50,
			FromDate: &from,
			ToDate:   &to,
		},
	})
	if err != nil {
		log.Fatal(err)
	}
	fmt.Println(extractions)

	// Filter by metadata
	extractions, err = client.Extractions.List(ctx, &retab.ListExtractionsParams{
		Metadata: map[string]string{"source": "docs"},
	})
	if err != nil {
		log.Fatal(err)
	}
	fmt.Println(extractions)
}
```

```rust Rust
use retab::enums::ExtractionsOrder;
use retab::resources::extractions::ListParams;
use retab::Retab;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let client = Retab::new(std::env::var("RETAB_API_KEY")?);

    // List recent extractions
    let _extractions = client
        .extractions()
        .list(ListParams {
            limit: Some(10),
            order: Some(ExtractionsOrder::Desc),
            ..Default::default()
        })
        .await?;

    // Filter by date range
    let _ranged = client
        .extractions()
        .list(ListParams {
            from_date: Some("2024-01-01".into()),
            to_date: Some("2024-12-31".into()),
            limit: Some(50),
            ..Default::default()
        })
        .await?;

    // Filter by metadata
    let _by_metadata = client
        .extractions()
        .list(ListParams {
            metadata: Some(r#"{"source":"docs"}"#.into()),
            ..Default::default()
        })
        .await?;
    Ok(())
}
```
