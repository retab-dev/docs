---
title: "Update Table"
openapi: "PATCH /v1/tables/{table_id}"
---

Update table name or metadata. This does not mutate rows, columns, cells, or the backing CSV contents.

<RequestExample>

```python Python
from retab import Retab

client = Retab()

tables = client.tables.update_table(
    table_id="workflow_table_123",
    name="Carriers",
    metadata={"region": "eu"},
)
print(tables)
```

```go Go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	name := "Carriers"
	metadata := map[string]interface{}{"region": "eu"}
	tables, err := client.Tables.Update(ctx, "workflow_table_123", &retab.TablesUpdateParams{
		Name:     &name,
		Metadata: &metadata,
	})
	if err != nil {
		log.Fatal(err)
	}
	fmt.Println(*tables)
}
```

```rust Rust
use retab::Retab;
use retab::models::UpdateWorkflowTableRequest;
use retab::resources::tables::UpdateTableParams;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let client = Retab::new(std::env::var("RETAB_API_KEY")?);

    let tables = client
        .tables()
        .update_table(
            "workflow_table_123",
            UpdateTableParams::new(UpdateWorkflowTableRequest {
                name: Some("Carriers".to_string()),
                ..Default::default()
            }),
        )
        .await?;
    println!("{:?}", tables);
    Ok(())
}
```
