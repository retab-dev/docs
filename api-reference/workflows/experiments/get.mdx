---
title: 'Get Experiment'
openapi: 'GET /v1/workflows/{workflow_id}/experiments/{experiment_id}'
---

Fetch one experiment by id. The response includes the latest run status,
score, staleness, and schema-drift information — the same shape as the
listing endpoint.

<RequestExample>

```python python
from retab import Retab

client = Retab()

experiment = client.workflows.experiments.get(
    workflow_id="wf_abc123",
    experiment_id="exp_abc",
)
```

```javascript javascript
import { Retab } from '@retab/node';

const client = new Retab();

const experiment = await client.workflows.experiments.get(
  "wf_abc123",
  "exp_abc",
);
```

```go go
package main

import (
	"encoding/json"
	"fmt"
	"io"
	"log"
	"net/http"
	"os"
)

func main() {
	request, err := http.NewRequest(
		http.MethodGet,
		"https://api.retab.com/v1/workflows/wf_abc123/experiments/exp_abc",
		nil,
	)
	if err != nil {
		log.Fatal(err)
	}
	request.Header.Set("Api-Key", os.Getenv("RETAB_API_KEY"))

	response, err := http.DefaultClient.Do(request)
	if err != nil {
		log.Fatal(err)
	}
	defer response.Body.Close()

	responseBody, err := io.ReadAll(response.Body)
	if err != nil {
		log.Fatal(err)
	}
	if response.StatusCode >= 400 {
		log.Fatalf("request failed: %s", responseBody)
	}

	var experiment map[string]any
	if err := json.Unmarshal(responseBody, &experiment); err != nil {
		log.Fatal(err)
	}
	fmt.Println(experiment["id"])
}
```
