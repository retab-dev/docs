---
title: 'List Experiments'
openapi: 'GET /v1/workflows/{workflow_id}/experiments'
---

List all experiments attached to a workflow, with the latest run's
status, score, staleness flag, and merkle-based schema drift.

<RequestExample>

```python python
from retab import Retab

client = Retab()

experiments = client.workflows.experiments.list(workflow_id="wf_abc123")
for exp in experiments:
    print(exp.id, exp.name, exp.status, exp.score)
```

```javascript javascript
import { Retab } from '@retab/node';

const client = new Retab();

const experiments = await client.workflows.experiments.list("wf_abc123");
```

```go go
package main

import (
	"encoding/json"
	"fmt"
	"io"
	"log"
	"net/http"
	"os"
)

func main() {
	request, err := http.NewRequest(
		http.MethodGet,
		"https://api.retab.com/v1/workflows/wf_abc123/experiments",
		nil,
	)
	if err != nil {
		log.Fatal(err)
	}
	request.Header.Set("Api-Key", os.Getenv("RETAB_API_KEY"))

	response, err := http.DefaultClient.Do(request)
	if err != nil {
		log.Fatal(err)
	}
	defer response.Body.Close()

	responseBody, err := io.ReadAll(response.Body)
	if err != nil {
		log.Fatal(err)
	}
	if response.StatusCode >= 400 {
		log.Fatalf("request failed: %s", responseBody)
	}

	var experiments []map[string]any
	if err := json.Unmarshal(responseBody, &experiments); err != nil {
		log.Fatal(err)
	}
	for _, experiment := range experiments {
		fmt.Println(experiment["id"], experiment["name"], experiment["status"], experiment["score"])
	}
}
```
