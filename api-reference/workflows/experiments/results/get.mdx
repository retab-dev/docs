---
title: "Get Experiment Run Result"
openapi: "GET /v1/workflows/experiments/results/{result_id}"
---

Fetch one experiment result by its opaque `result_id`. Use this to refresh a
single document result without re-listing every result for the parent
experiment run.

<RequestExample>

```python python
from retab import Retab

client = Retab()

result = client.workflows.experiments.runs.results.get("expresult_1")

print(result.document_id)
print(result.lifecycle.status)
print(result.artifact)
```

```javascript javascript
const resultId = "expresult_1";

const response = await fetch(
  `https://api.retab.com/v1/workflows/experiments/results/${resultId}`,
  {
    headers: {
      "Api-Key": process.env.RETAB_API_KEY,
    },
  },
);

if (!response.ok) {
  throw new Error(`Retab request failed: ${response.status}`);
}

const result = await response.json();
console.log(result.document_id);
console.log(result.lifecycle.status);
console.log(result.artifact);
```

```go go
package main

import (
	"encoding/json"
	"fmt"
	"log"
	"net/http"
	"os"
)

func main() {
	resultID := "expresult_1"
	req, err := http.NewRequest(
		http.MethodGet,
		"https://api.retab.com/v1/workflows/experiments/results/"+resultID,
		nil,
	)
	if err != nil {
		log.Fatal(err)
	}
	req.Header.Set("Api-Key", os.Getenv("RETAB_API_KEY"))

	resp, err := http.DefaultClient.Do(req)
	if err != nil {
		log.Fatal(err)
	}
	defer resp.Body.Close()

	if resp.StatusCode < 200 || resp.StatusCode >= 300 {
		log.Fatalf("Retab request failed: %s", resp.Status)
	}

	var result map[string]any
	if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
		log.Fatal(err)
	}

	fmt.Println(result["document_id"])
	fmt.Println(result["lifecycle"])
	fmt.Println(result["artifact"])
}
```
