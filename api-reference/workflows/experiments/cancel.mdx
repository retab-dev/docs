---
title: 'Cancel Experiment'
openapi: 'POST /v1/workflows/{workflow_id}/experiments/{experiment_id}/cancel'
---

Cancel the latest pending or running experiment run. Returns `404` if no
in-flight run exists.

Useful before [Delete Experiment](/api-reference/workflows/experiments/delete),
which fails with `409` while a run is in progress.

<RequestExample>

```python python
from retab import Retab

client = Retab()

result = client.workflows.experiments.cancel(
    workflow_id="wf_abc123",
    experiment_id="exp_abc",
)
print(result["status"], result["run_id"])
```

package main

import (
	"encoding/json"
	"fmt"
	"io"
	"log"

	request, err := http.NewRequest(
		http.MethodPost,

		nil,
	)
	if err != nil {
		log.Fatal(err)
	}
	request.Header.Set("Api-Key", os.Getenv("RETAB_API_KEY"))

	response, err := http.DefaultClient.Do(request)
	if err != nil {
		log.Fatal(err)
	}
	defer response.Body.Close()

	responseBody, err := io.ReadAll(response.Body)
	if err != nil {
		log.Fatal(err)
	}
	if response.StatusCode >= 400 {
		log.Fatalf("request failed: %s", responseBody)
	}

	var result map[string]any
	if err := json.Unmarshal(responseBody, &result); err != nil {
		log.Fatal(err)
	}
	fmt.Println(result["status"], result["run_id"])
}
