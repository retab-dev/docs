---
title: "Get Review"
openapi: "GET /v1/workflows/reviews/{review_id}"
---

Get one full review by opaque review id.

<RequestExample>

```python python
from retab import Retab

client = Retab()

review = client.workflows.reviews.get("rev_D4J7WZBRV4H7C2SKQJ4NP6W2EY")

print(review.id, review.workflow_run_id, review.block_id)
if review.decision:
    print(review.decision.verdict, review.decision.version_id)
```

```javascript javascript
import { Retab } from "@retab/node";

const client = new Retab();

const review = await client.workflows.reviews.get(
  "rev_D4J7WZBRV4H7C2SKQJ4NP6W2EY",

```

```go go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	review, err := client.Workflows.Reviews.Get(ctx, "rev_D4J7WZBRV4H7C2SKQJ4NP6W2EY")
	if err != nil {
		log.Fatal(err)
	}

	fmt.Println(review.ID, review.WorkflowRunID, review.BlockID)
	if review.Decision != nil {
		fmt.Println(review.Decision.Verdict, review.Decision.VersionID)
	}
}
```
