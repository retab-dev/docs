---
title: "Get Review"
openapi: "GET /v1/workflows/reviews/{review_id}"
---

Get one full review by opaque review id.

<RequestExample>

```python Python
from retab import Retab

client = Retab()

review = client.workflows.reviews.get("rev_D4J7WZBRV4H7C2SKQJ4NP6W2EY")

print(review.id, review.run_id, review.block_id)
if review.decision:
    print(review.decision.verdict, review.decision.version_id)
```

```go Go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func ptr[T any](v T) *T { return &v }

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	review, err := client.Workflows.Reviews.Get(ctx, "rev_D4J7WZBRV4H7C2SKQJ4NP6W2EY")
	if err != nil {
		log.Fatal(err)
	}

	fmt.Println(review.ID, review.RunID, review.BlockID)
	if review.Decision != nil {
		fmt.Println(review.Decision.Verdict, review.Decision.VersionID)
	}
}
```

```rust Rust
use retab::Retab;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let client = Retab::new(std::env::var("RETAB_API_KEY")?);

    let review = client
        .workflows().reviews()
        .get("rev_D4J7WZBRV4H7C2SKQJ4NP6W2EY")
        .await?;
    println!("{} {} {}", review.id, review.run_id, review.block_id);
    if let Some(decision) = review.decision.as_ref() {
        println!("{:?} {}", decision.verdict, decision.version_id);
    }
    Ok(())
}
```
