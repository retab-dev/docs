---
title: "Get Version"
openapi: "GET /v1/workflows/reviews/versions/{version_id}"
---

Fetch one review version by its content-addressed `rvr_…` id.

<RequestExample>

```python Python
from retab import Retab

client = Retab()

version = client.workflows.reviews.versions.get(
    "rvr_PFBGO3R6T3PG5O37U4Y3NDXWAM",
)
print(version.snapshot)
```

```go Go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func ptr[T any](v T) *T { return &v }

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	version, err := client.Workflows.Reviews.Versions.Get(ctx, "rvr_PFBGO3R6T3PG5O37U4Y3NDXWAM")
	if err != nil {
		log.Fatal(err)
	}

	fmt.Println(version.Snapshot)
}
```
