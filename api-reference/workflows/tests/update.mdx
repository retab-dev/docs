---
title: "Update Workflow Test"
openapi: "PATCH /v1/workflows/tests/{test_id}"
---

Patch the name, assertion, and/or source of a workflow test. Every field is
optional in the request body — fields you omit are left untouched. Setting
`assertion: null` is rejected explicitly (use [Delete Workflow Test](./delete) if
you want to remove the test entirely).

Updating `source` rebinds where the inputs come from:

- `{ type: "manual", handle_inputs: ... }` — switch to hand-written inputs.
- `{ type: "run_step", run_id: ..., step_id: ... }` — re-capture inputs from
  a different workflow run / step.

The response is the full updated `WorkflowTest`.

<RequestExample>

```python python
from retab import Retab

client = Retab()

# Rename only.
test = client.workflows.tests.update(
    test_id="wfnodetest_hsLEQiM61ez9Piv147MWk",
    name="Q1 invoice — vendor name check",
)

# Replace the assertion.
test = client.workflows.tests.update(
    test_id="wfnodetest_hsLEQiM61ez9Piv147MWk",
    assertion={
        "target": {"output_handle_id": "output-json-0", "path": "vendor.name"},
        "condition": {"kind": "matches_regex", "expected": r"^Acme.*"},
    },
)

# Re-capture inputs from a different run.
test = client.workflows.tests.update(
    test_id="wfnodetest_hsLEQiM61ez9Piv147MWk",
    source={
        "type": "run_step",
        "run_id": "wfrun_new789",
        "step_id": "block_extract_invoice",
    },
)
```

```go go
package main

import (
	"context"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	testID := "wfnodetest_hsLEQiM61ez9Piv147MWk"

	// Rename only.
	if _, err := client.Workflows.Tests.Update(ctx, testID, retab.UpdateWorkflowTestRequest{
		Name: "Q1 invoice — vendor name check",
	}); err != nil {
		log.Fatal(err)
	}

	// Replace the assertion.
	if _, err := client.Workflows.Tests.Update(ctx, testID, retab.UpdateWorkflowTestRequest{
		Assertion: retab.Resource{
			"target":    map[string]any{"output_handle_id": "output-json-0", "path": "vendor.name"},
			"condition": map[string]any{"kind": "matches_regex", "expected": "^Acme.*"},
		},
	}); err != nil {
		log.Fatal(err)
	}

	// Re-capture inputs from a different run.
	if _, err := client.Workflows.Tests.Update(ctx, testID, retab.UpdateWorkflowTestRequest{
		Source: retab.Resource{
			"type":    "run_step",
			"run_id":  "wfrun_new789",
			"step_id": "block_extract_invoice",
		},
	}); err != nil {
		log.Fatal(err)
	}
}
```
