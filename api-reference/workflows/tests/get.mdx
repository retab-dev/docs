---
title: "Get Workflow Test"
openapi: "GET /v1/workflows/{workflow_id}/tests/{test_id}"
---

Fetch a single workflow test by id. The response refreshes the test's
`validation_status` and schema-drift state against the current workflow draft
before returning, so the caller always sees the latest staleness.

<RequestExample>

```python python
from retab import Retab

client = Retab()

test = client.workflows.tests.get(
    workflow_id="wf_abc123xyz",
    test_id="wfnodetest_hsLEQiM61ez9Piv147MWk",
)

print(test.assertion.condition)
print(test.schema_drift)
```

```javascript javascript
import { Retab } from "@retab/node";

const client = new Retab();

const test = await client.workflows.tests.get({
  workflowId: "wf_abc123xyz",
  testId: "wfnodetest_hsLEQiM61ez9Piv147MWk",
});

console.log(test.assertion.condition);
console.log(test.schema_drift);
```

```go go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	test, err := client.Workflows.Tests.Get(ctx, "wf_abc123xyz", "wfnodetest_hsLEQiM61ez9Piv147MWk")
	if err != nil {
		log.Fatal(err)
	}

	if assertion, ok := (*test)["assertion"].(map[string]any); ok {
		fmt.Println(assertion["condition"])
	}
	fmt.Println((*test)["schema_drift"])
}
```
