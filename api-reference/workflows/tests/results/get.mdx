---
title: "Get Test Run Result"
openapi: "GET /v1/workflows/tests/results/{result_id}"
---

Fetch one workflow-test result by its opaque `result_id`. Use this to refresh a
single result row without re-listing every result for the parent test run.

<RequestExample>

```python python
from retab import Retab

client = Retab()

result = client.workflows.tests.runs.results.get("wfnodetestrun_a")

print(result.test_id)
print(result.lifecycle.status)
print(result.verdict)
```

```javascript javascript
const resultId = "wfnodetestrun_a";

const response = await fetch(
  `https://api.retab.com/v1/workflows/tests/results/${resultId}`,
  {
    headers: {
      "Api-Key": process.env.RETAB_API_KEY,
    },
  },
);

if (!response.ok) {
  throw new Error(`Retab request failed: ${response.status}`);
}

const result = await response.json();
console.log(result.test_id);
console.log(result.lifecycle.status);
console.log(result.verdict);
```

```go go
package main

import (
	"encoding/json"
	"fmt"
	"log"
	"net/http"
	"os"
)

func main() {
	resultID := "wfnodetestrun_a"
	req, err := http.NewRequest(
		http.MethodGet,
		"https://api.retab.com/v1/workflows/tests/results/"+resultID,
		nil,
	)
	if err != nil {
		log.Fatal(err)
	}
	req.Header.Set("Api-Key", os.Getenv("RETAB_API_KEY"))

	resp, err := http.DefaultClient.Do(req)
	if err != nil {
		log.Fatal(err)
	}
	defer resp.Body.Close()

	if resp.StatusCode < 200 || resp.StatusCode >= 300 {
		log.Fatalf("Retab request failed: %s", resp.Status)
	}

	var result map[string]any
	if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
		log.Fatal(err)
	}

	fmt.Println(result["test_id"])
	fmt.Println(result["lifecycle"])
	fmt.Println(result["verdict"])
}
```

curl -X 'GET' \
