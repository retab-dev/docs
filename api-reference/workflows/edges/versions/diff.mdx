---
title: "Diff Edge Versions"
openapi: "GET /v1/workflows/edges/versions/diff"
---

Compare two immutable workflow edge versions.

<RequestExample>

```python Python
from retab import Retab

client = Retab()

diff = client.workflows.edges.list_diff(
    from_edge_version_id="egv_before",
    to_edge_version_id="egv_after",
)
print(diff)
```

```go Go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	diff, err := client.Workflows.Edges.ListDiff(ctx, &retab.WorkflowEdgesListDiffParams{
		FromEdgeVersionID: "egv_before",
		ToEdgeVersionID:   "egv_after",
	})
	if err != nil {
		log.Fatal(err)
	}
	fmt.Println(*diff)
}
```

```rust Rust
use retab::Retab;
use retab::resources::workflow_edges::ListDiffParams;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let client = Retab::new(std::env::var("RETAB_API_KEY")?);

    let diff = client
        .workflows()
        .edges()
        .list_diff(ListDiffParams::new("egv_before", "egv_after"))
        .await?;
    println!("{:?}", diff);
    Ok(())
}
```
