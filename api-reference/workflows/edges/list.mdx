---
title: 'List Edges'
openapi: 'GET /v1/workflows/{workflow_id}/edges'
---

List edges in a workflow's current draft. Optionally filter to edges incident to a specific block by passing `source_block` or `target_block`.

For the full graph in a single call, use [Get Workflow Entities](/api-reference/workflows/get-entities).

<RequestExample>

```python python
from retab import Retab

client = Retab()

# All edges
edges = client.workflows.edges.list("wf_abc123xyz")

# Outgoing edges of one block
outgoing = client.workflows.edges.list(
    "wf_abc123xyz",
    source_block="extract-1",
)

# Incoming edges of one block
incoming = client.workflows.edges.list(
    "wf_abc123xyz",
    target_block="hil-1",
)

for edge in edges:
    print(f"{edge.source_block}/{edge.source_handle} -> {edge.target_block}/{edge.target_handle}")
```

```go go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	// All edges
	edges, err := client.Workflows.Edges.List(ctx, "wf_abc123xyz", nil)
	if err != nil {
		log.Fatal(err)
	}

	// Outgoing edges of one block
	outgoing, err := client.Workflows.Edges.List(ctx, "wf_abc123xyz", &retab.ListWorkflowEdgesParams{
		SourceBlock: "extract-1",
	})
	if err != nil {
		log.Fatal(err)
	}

	// Incoming edges of one block
	incoming, err := client.Workflows.Edges.List(ctx, "wf_abc123xyz", &retab.ListWorkflowEdgesParams{
		TargetBlock: "hil-1",
	})
	if err != nil {
		log.Fatal(err)
	}

	for _, edge := range edges {
		fmt.Printf("%s/%s -> %s/%s\n", edge.SourceBlock, edge.SourceHandle, edge.TargetBlock, edge.TargetHandle)
	}
	_, _ = outgoing, incoming
}
```

```bash curl

  -H 'Api-Key: <your-api-key>'

curl -X 'GET' \
  'https://api.retab.com/v1/workflows/wf_abc123xyz/edges?source_block=extract-1' \
  -H 'Api-Key: <your-api-key>'
```
