---
title: "Get Block"
openapi: "GET /v1/workflows/blocks/{block_id}"
---

Get a single authored block from the workflow's current draft.

<RequestExample>

```python python
from retab import Retab

client = Retab()

block = client.workflows.blocks.get(
    "extract-1",
)

print(block.type, block.label)
print(block.config)
```

```typescript TypeScript
import { Retab } from "@retab/node";

const client = new Retab({ apiKey: process.env.RETAB_API_KEY });

const block = await client.workflows.blocks.get("extract-1");

console.log(block.type, block.label);
console.log(block.config);
```

```go go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	block, err := client.Workflows.Blocks.Get(ctx, "extract-1")
	if err != nil {
		log.Fatal(err)
	}

	fmt.Println(block.Type, block.Label)
	fmt.Println(block.Config)
}
```

```rust rust
use retab::resources::workflow_blocks::GetParams;
use retab::Retab;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let client = Retab::new(std::env::var("RETAB_API_KEY")?);

    let block = client
        .workflow_blocks()
        .get("extract-1", GetParams::default())
        .await?;

    println!("{} {}", block.type_, block.label.as_deref().unwrap_or(""));
    println!("{:?}", block.config);
    Ok(())
}
```
