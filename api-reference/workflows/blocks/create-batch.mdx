---
title: 'Create Blocks (batch)'
openapi: 'POST /v1/workflows/{workflow_id}/blocks/batch'
---

Create multiple blocks in a single request. Useful when scaffolding a workflow programmatically — pair it with [Create Edges (batch)](/api-reference/workflows/edges/create-batch) to build the whole graph in two calls.

Each entry has the same shape as the body of [Create Block](/api-reference/workflows/blocks/create). The whole batch is rejected if any entry is invalid (duplicate ID, input block inside a container, unknown `parent_id`, …).

<RequestExample>

```python python
from retab import Retab

client = Retab()

blocks = client.workflows.blocks.create_batch(
    workflow_id="wf_abc123xyz",
    blocks=[
        {
            "id": "start-1",
            "type": "start-document",
            "label": "Invoice Input",
            "position_x": 0,
            "position_y": 0,
        },
        {
            "id": "extract-1",
            "type": "extract",
            "label": "Extract Invoice Fields",
            "position_x": 320,
            "position_y": 0,
            "config": {
                "model": "gpt-5",
                "json_schema": {
                    "type": "object",
                    "required": ["total"],
                    "properties": {"total": {"type": "number"}},
                },
                "hil": {
                    "predicate": {"kind": "field_confidence_lt", "path": "total", "threshold": 0.75},
                    "skip_in_test_mode": false,
                },
            },
        },
    ],
)

for block in blocks:
    print(block.id, block.type)
```

```javascript javascript
      label: "Extract Invoice Fields",

        model: "retab-small",

        hil: {
          predicate: { kind: "field_confidence_lt", path: "total", threshold: 0.75 },
          skip_in_test_mode: false,
        },
      },
    },
]);

for (const block of blocks) {
  console.log(block.id, block.type);
}
```

```go go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	blocks, err := client.Workflows.Blocks.CreateBatch(ctx, "wf_abc123xyz", []retab.WorkflowBlockCreateRequest{
		{ID: "start-1", Type: "start", Label: "Invoice Input", PositionX: 0, PositionY: 0},
		{
			ID:        "extract-1",
			Type:      "extract",
			Label:     "Extract Invoice Fields",
			PositionX: 320,
			PositionY: 0,
			Config: map[string]any{
				"model": "gpt-5",
				"json_schema": map[string]any{
					"type":       "object",
					"required":   []any{"total"},
					"properties": map[string]any{"total": map[string]any{"type": "number"}},
				},
				"hil": map[string]any{
					"predicate": map[string]any{"kind": "field_confidence_lt", "path": "total", "threshold": 0.75},
					"skip_in_test_mode": false,
				},
			},
		},
	})
	if err != nil {
		log.Fatal(err)
	}

	for _, block := range blocks {
		fmt.Println(block.ID, block.Type)
	}
}
```
