---
title: "Diff Block Versions"
openapi: "GET /v1/workflows/blocks/versions/diff"
---

Compare two immutable workflow block versions.

<RequestExample>

```python Python
from retab import Retab

client = Retab()

diff = client.workflows.blocks.list_diff(
    from_block_version_id="bkv_before",
    to_block_version_id="bkv_after",
)
print(diff)
```

```go Go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	diff, err := client.Workflows.Blocks.ListDiff(ctx, &retab.WorkflowBlocksListDiffParams{
		FromBlockVersionID: "bkv_before",
		ToBlockVersionID:   "bkv_after",
	})
	if err != nil {
		log.Fatal(err)
	}
	fmt.Println(*diff)
}
```

```rust Rust
use retab::resources::workflow_blocks::ListDiffParams;
use retab::Retab;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let client = Retab::new(std::env::var("RETAB_API_KEY")?);

    let diff = client
        .workflows()
        .blocks()
        .list_diff(ListDiffParams::new("bkv_before", "bkv_after"))
        .await?;
    println!("{:?}", diff);
    Ok(())
}
```
