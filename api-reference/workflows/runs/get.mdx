---
title: 'Get Run'
openapi: 'GET /v1/workflows/runs/{run_id}'
---

Get a single workflow run by ID. Use this endpoint to check `lifecycle.status` for a running workflow or retrieve the results of a completed workflow.

Run timing includes timestamps, not precomputed duration fields. Derive duration from
`timing.completed_at - timing.started_at` when both timestamps are present.

Run responses do not embed step records. Use [List Steps](/api-reference/workflows/runs/steps/list)
for per-block status and `started_at` / `completed_at`, and use
[Get Step](/api-reference/workflows/runs/steps/get) for typed handle inputs
and outputs. For persisted step records such as HIL evaluations, function
invocations, or API call traces, use [List Artifacts](/api-reference/workflows/artifacts/list).

<RequestExample>

```python python
from retab import Retab

client = Retab()

run = client.workflows.runs.get("run_abc123xyz")

print(f"Lifecycle: {run.lifecycle.status}")
if run.lifecycle.status == "completed":
    if run.timing.started_at and run.timing.completed_at:
        duration_ms = int((run.timing.completed_at - run.timing.started_at).total_seconds() * 1000)
        print(f"Duration: {duration_ms}ms")
    for step_summary in client.workflows.runs.steps.list(run.id):
        step = client.workflows.runs.steps.get(run.id, step_summary.block_id)
        if step.handle_outputs:
            print(f"{step.block_id}: {step.handle_outputs}")
        if step.artifact:
            artifact = client.workflows.artifacts.get(step.artifact)
            print(f"{step.block_id} artifact: {artifact.operation}")
```

```javascript javascript
import { Retab } from '@retab/node';

const client = new Retab();

const run = await client.workflows.runs.get("run_abc123xyz");

console.log(`Lifecycle: ${run.lifecycle.status}`);
if (run.lifecycle.status === "completed") {
    if (run.timing.started_at && run.timing.completed_at) {
        const durationMs = new Date(run.timing.completed_at).getTime() - new Date(run.timing.started_at).getTime();
        console.log(`Duration: ${durationMs}ms`);
    }
    const stepSummaries = await client.workflows.runs.steps.list(run.id);
    for (const stepSummary of stepSummaries) {
        const step = await client.workflows.runs.steps.get(run.id, stepSummary.block_id);
        if (Object.keys(step.handle_outputs).length > 0) {
            console.log(`${step.block_id}:`, step.handle_outputs);
        }
        if (step.artifact) {
            const artifact = await client.workflows.artifacts.get(step.artifact);
            console.log(`${step.block_id} artifact:`, artifact.operation);
        }
    }
}
```

```go go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	run, err := client.Workflows.Runs.Get(ctx, "run_abc123xyz")
	if err != nil {
		log.Fatal(err)
	}

	fmt.Printf("Lifecycle: %s\n", run.Lifecycle.Status)
	if run.Lifecycle.Status == "completed" {
		stepSummaries, err := client.Workflows.Runs.Steps.List(ctx, run.ID)
		if err != nil {
			log.Fatal(err)
		}
		for _, summary := range stepSummaries {
			step, err := client.Workflows.Runs.Steps.Get(ctx, run.ID, summary.BlockID)
			if err != nil {
				log.Fatal(err)
			}
			if len(step.HandleOutputs) > 0 {
				fmt.Printf("%s: %v\n", step.BlockID, step.HandleOutputs)
			}
		}
	}
}
```
