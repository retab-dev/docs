---
title: 'Get Run'
openapi: 'GET /v1/workflows/runs/{run_id}'
---

Get a single workflow run by ID. Use this endpoint to check `lifecycle.kind` for a running workflow or retrieve the results of a completed workflow.

`run.steps` contains execution status for each block. For typed handle inputs
and outputs, use [Get Step](/api-reference/workflows/runs/steps/get). For
persisted step records such as HIL evaluations, function invocations, or API
call traces, use [List Artifacts](/api-reference/workflows/artifacts/list).

<RequestExample>

```python python
from retab import Retab

client = Retab()

run = client.workflows.runs.get("run_abc123xyz")

print(f"Lifecycle: {run.lifecycle.kind}")
if run.lifecycle.kind == "completed":
    print(f"Duration: {run.timing.duration_ms}ms")
    for step_summary in client.workflows.runs.steps.list(run.id):
        step = client.workflows.runs.steps.get(run.id, step_summary.block_id)
        if step.handle_outputs:
            print(f"{step.block_id}: {step.handle_outputs}")
        if step.artifact:
            artifact = client.workflows.artifacts.get(step.artifact)
            print(f"{step.block_id} artifact: {artifact.operation}")
```

```javascript javascript

            console.log(`${step.block_id}:`, step.handle_outputs);
        }
        if (step.artifact) {
            const artifact = await client.workflows.artifacts.get(step.artifact);
            console.log(`${step.block_id} artifact:`, artifact.operation);
        }
    }
}
```

```go go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	run, err := client.Workflows.Runs.Get(ctx, "run_abc123xyz")
	if err != nil {
		log.Fatal(err)
	}

	fmt.Printf("Lifecycle: %s\n", run.Lifecycle.Kind)
	if run.Lifecycle.Kind == "completed" {
		stepSummaries, err := client.Workflows.Runs.Steps.List(ctx, run.ID)
		if err != nil {
			log.Fatal(err)
		}
		for _, summary := range stepSummaries {
			step, err := client.Workflows.Runs.Steps.Get(ctx, run.ID, summary.BlockID)
			if err != nil {
				log.Fatal(err)
			}
			if len(step.HandleOutputs) > 0 {
				fmt.Printf("%s: %v\n", step.BlockID, step.HandleOutputs)
			}
		}
	}
}
```
