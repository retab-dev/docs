---
title: 'Get Workflow'
openapi: 'GET /v1/workflows/{workflow_id}'
---

Get a single workflow by ID.

<RequestExample>

```python python
from retab import Retab

client = Retab()

workflow = client.workflows.get("workflow_abc123xyz")

print(workflow.name)
print(workflow.published.version_id if workflow.published else None)
print(workflow.updated_at)
```

```javascript javascript
import { Retab } from '@retab/node';

const client = new Retab();

const workflow = await client.workflows.get("workflow_abc123xyz");

console.log(workflow.name);
console.log(workflow.published?.version_id);
console.log(workflow.updated_at);
```

```go go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	workflow, err := client.Workflows.Get(ctx, "workflow_abc123xyz")
	if err != nil {
		log.Fatal(err)
	}

	fmt.Println(workflow.Name)
	if workflow.Published != nil {
		fmt.Println(workflow.Published.VersionID)
	}
	fmt.Println(workflow.UpdatedAt)
}
```
