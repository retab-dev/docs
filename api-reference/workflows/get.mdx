---
title: "Get Workflow"
openapi: "GET /v1/workflows/{workflow_id}"
---

Get a single workflow by ID.

<RequestExample>

```python Python
from retab import Retab

client = Retab()

workflow = client.workflows.get("workflow_abc123xyz")

print(workflow.name)
print(workflow.published.version_id if workflow.published else None)
print(workflow.updated_at)
```

```go Go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func ptr[T any](v T) *T { return &v }

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	workflow, err := client.Workflows.Get(ctx, "workflow_abc123xyz")
	if err != nil {
		log.Fatal(err)
	}

	fmt.Println(workflow.Name)
	if workflow.Published != nil {
		fmt.Println(workflow.Published.VersionID)
	}
	fmt.Println(workflow.UpdatedAt)
}
```

```rust Rust
use retab::Retab;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let client = Retab::new(std::env::var("RETAB_API_KEY")?);

    let workflow = client.workflows().get("workflow_abc123xyz").await?;

    println!("{}", workflow.name.as_deref().unwrap_or(""));
    if let Some(published) = &workflow.published {
        println!("{}", published.version_id.as_deref().unwrap_or(""));
    }
    println!("{}", workflow.updated_at);
    Ok(())
}
```
