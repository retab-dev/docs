---
title: "Diff Workflow Versions"
openapi: "GET /v1/workflows/versions/diff"
---

Compare two immutable workflow graph versions.

<RequestExample>

```python Python
from retab import Retab

client = Retab()

diff = client.workflows.list_diff(
    workflow_id="wf_abc123",
    from_workflow_version_id="wfv_before",
    to_workflow_version_id="wfv_after",
)
print(diff)
```


```go Go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	diff, err := client.Workflows.ListDiff(ctx, &retab.WorkflowsListDiffParams{
		WorkflowID:            "wf_abc123",
		FromWorkflowVersionID: "wfv_before",
		ToWorkflowVersionID:   "wfv_after",
	})
	if err != nil {
		log.Fatal(err)
	}
	fmt.Println(*diff)
}
```

```rust Rust
use retab::Retab;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let client = Retab::new(std::env::var("RETAB_API_KEY")?);

    let diff = client
        .workflows()
        .list_diff(ListDiffParams::new("wf_abc123", "wfv_before", "wfv_after"))
        .await?;
    println!("{:?}", diff);
    Ok(())
}
```
