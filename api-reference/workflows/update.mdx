---
title: 'Update Workflow'
openapi: 'PATCH /v1/workflows/{workflow_id}'
---

Update workflow metadata. Only the fields you supply are touched — omit a field to leave it unchanged.

<RequestExample>

```python python
from retab import Retab

client = Retab()

workflow = client.workflows.update(
    "wf_abc123xyz",
    name="Invoice Processing v2",
    description="Extract invoice fields, validate, and route for review",
)
```

```javascript javascript

const client = new Retab();

const workflow = await client.workflows.update("wf_abc123xyz", {
  name: "Invoice Processing v2",
  description: "Extract invoice fields, validate, and route for review",
});
```

```go go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	name := "Invoice Processing v2"
	description := "Extract invoice fields, validate, and route for review"
	workflow, err := client.Workflows.Update(ctx, "wf_abc123xyz", retab.UpdateWorkflowRequest{
		Name:        &name,
		Description: &description,
	})
	if err != nil {
		log.Fatal(err)
	}

	fmt.Println(workflow.Name)
}
```

```bash curl
curl -X 'PATCH' \

  }'
```
