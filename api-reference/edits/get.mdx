```go Go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func ptr[T any](v T) *T { return &v }

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	edit, err := client.Edits.Get(ctx, "edit_01G34H8J2K")
	if err != nil {
		log.Fatal(err)
	}
	fmt.Println(edit.Output.FormData)
}
```
