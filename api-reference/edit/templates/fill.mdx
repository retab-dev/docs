title: 'Fill Template'
openapi: 'POST /v1/edit/templates/fill'
---

<RequestExample>

```python python
from retab import Retab
import base64

client = Retab()

# Fill a form using a pre-defined template
response = client.edit.templates.fill(
    template_id="edittplt_abc123",
    instructions="""
    Full Name: Jane Smith
    Date of Birth: March 15, 1985
    Email: jane.smith@example.com
    Phone: (555) 123-4567
    """,
    model="retab-small"
)

print(f"Filled {len(response.form_data)} fields")

# Save the filled PDF
if response.filled_document:
    pdf_bytes = base64.b64decode(response.filled_document.url.split(",")[1])
    with open("filled_form.pdf", "wb") as f:
        f.write(pdf_bytes)
    print("Filled form saved!")
```

```javascript javascript
import { Retab } from '@retab/node';
import fs from 'fs';

const client = new Retab();

// Fill a form using a pre-defined template
const response = await client.edit.templates.fill({
    template_id: "edittplt_abc123",
    instructions: `
        Full Name: Jane Smith
        Date of Birth: March 15, 1985
        Email: jane.smith@example.com
        Phone: (555) 123-4567
    `,
    model: "retab-small"
});

console.log(`Filled ${response.form_data.length} fields`);

// Save the filled PDF
const base64Content = response.filled_document.url.split(",")[1];
fs.writeFileSync("filled_form.pdf", Buffer.from(base64Content, "base64"));
console.log("Filled form saved!");
```

```bash curl
curl -X POST 'https://api.retab.com/v1/edit/templates/fill' \
  -H "Api-Key: $RETAB_API_KEY" \
  -H 'Content-Type: application/json' \
  -d '{
    "template_id": "edittplt_abc123",
    "instructions": "Full Name: Jane Smith\nDate of Birth: March 15, 1985\nEmail: jane.smith@example.com",
    "model": "retab-small"
  }'
```
