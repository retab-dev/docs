---
title: "Get Auth Status"
openapi: "GET /v1/auth/status"
---

Return the caller's resolved organization and environment context.

```python Python
from retab import Retab

client = Retab()

status = client.auth.list_status()
print(status.auth_method)
```

```typescript TypeScript
import { Retab } from "@retab/node";

const client = new Retab({ apiKey: process.env.RETAB_API_KEY });

const status = await client.auth.list_status();
console.log(status.authMethod);
```

```go Go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	status, err := client.Auth.ListStatus(context.Background())
	if err != nil {
		log.Fatal(err)
	}

	fmt.Println(status.AuthMethod)
}
```

```java Java
import com.retab.RetabClient;

public final class Example {
  public static void main(String[] args) throws Exception {
    RetabClient client = new RetabClient(System.getenv("RETAB_API_KEY"));

    var status = client.auth().listStatus();
    System.out.println(status.getAuthMethod());
  }
}
```
