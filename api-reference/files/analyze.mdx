---
title: "Analyze File"
openapi: "POST /v1/files/analyze"
---

Create an asynchronous file-analysis job for an uploaded file. Poll the returned job with `GET /v1/jobs/{job_id}` and include the response when you need the generated document blueprint.

<RequestExample>

```python Python
from retab import Retab
from retab.types.files import AnalyzeFileRequestMode

client = Retab()

job = client.files.create_analyze(
    "file_a1b2c3d4e5f6",
    mode=AnalyzeFileRequestMode.REASONING,
    intent="Identify fields and validations for an invoice workflow",
)
print(job.id)
```

```go Go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func ptr[T any](v T) *T { return &v }

func main() {
	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	job, err := client.Files.CreateAnalyze(context.Background(), &retab.FilesCreateAnalyzeParams{
		FileID: "file_a1b2c3d4e5f6",
		Mode: ptr(retab.AnalyzeFileRequestModeReasoning),
		Intent: ptr("Identify fields and validations for an invoice workflow"),
	})
	if err != nil {
		log.Fatal(err)
	}

	fmt.Println(job.ID)
}
```

```rust Rust
use retab::enums::AnalyzeFileRequestMode;
use retab::models::AnalyzeFileRequest;
use retab::resources::files::CreateAnalyzeParams;
use retab::Retab;

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let client = Retab::new(std::env::var("RETAB_API_KEY")?);

    let mut body = AnalyzeFileRequest::new("file_a1b2c3d4e5f6");
    body.mode = Some(AnalyzeFileRequestMode::Reasoning);
    body.intent = Some("Identify fields and validations for an invoice workflow".into());

    let job = client.files().create_analyze(CreateAnalyzeParams::new(body)).await?;
    println!("{}", job.id);
    Ok(())
}
```
