---
title: 'List Partitions'
openapi: 'GET /v1/partitions'
---

List persisted partitions for your organization, with id-based pagination and optional filename / date filtering.

<RequestExample>

```python python
from datetime import datetime
from retab import Retab

client = Retab()

partitions = client.partitions.list(limit=20, order="desc")
for partition in partitions.data:
    print(f"{partition.id}: {partition.file.filename}")

partitions = client.partitions.list(
    from_date=datetime(2024, 1, 1),
    to_date=datetime(2024, 12, 31),
    limit=50,
)
```


```go go
package main

import (
	"context"
	"fmt"
	"log"
	"time"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	partitions, err := client.Partitions.List(ctx, &retab.ListParams{Limit: 20, Order: "desc"})
	if err != nil {
		log.Fatal(err)
	}
	for _, partition := range partitions.Data {
		fmt.Printf("%v: %v\n", partition["id"], partition["file"])
	}

	from := time.Date(2024, 1, 1, 0, 0, 0, 0, time.UTC)
	to := time.Date(2024, 12, 31, 0, 0, 0, 0, time.UTC)
	ranged, err := client.Partitions.List(ctx, &retab.ListParams{
		FromDate: &from,
		ToDate:   &to,
		Limit:    50,
	})
	if err != nil {
		log.Fatal(err)
	}
	fmt.Println(ranged)
}
```
