---
title: "Create Partitions"
openapi: "POST /v1/partitions"
---

Partition a document into repeated chunks keyed by a value such as `invoice_number`, `policy_id`, or `claim_number`.

<RequestExample>

```python python
from retab import Retab

client = Retab()

response = client.partitions.create(
    document="invoice_batch.pdf",
    key="invoice_number",
    instructions="Return one chunk per invoice number and keep all pages for the same invoice together.",
    model="retab-small",
    n_consensus=3,
    allow_overlap=True,
    bust_cache=False,
)

for chunk in response.output:
    print(chunk.key, chunk.pages)
```

```go go
package main

import (
	"context"
	"fmt"
	"log"

	retab "github.com/retab-dev/retab/clients/go"
)

func main() {
	ctx := context.Background()

	client, err := retab.NewClient("")
	if err != nil {
		log.Fatal(err)
	}

	document, err := retab.InferMIMEData("invoice_batch.pdf")
	if err != nil {
		log.Fatal(err)
	}

	response, err := client.Partitions.Create(ctx, retab.PartitionCreateRequest{
		Document:     document,
		Key:          "invoice_number",
		Instructions: "Return one chunk per invoice number and keep all pages for the same invoice together.",
		Model:        "retab-small",
		NConsensus:   3,
		AllowOverlap: true,
		BustCache:    false,
	})
	if err != nil {
		log.Fatal(err)
	}

	if output, ok := (*response)["output"].([]any); ok {
		for _, chunk := range output {
			c, _ := chunk.(map[string]any)
			fmt.Println(c["key"], c["pages"])
		}
	}
}
```
